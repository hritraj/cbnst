CREATE TABLE Library (
    BookId VARCHAR(4) PRIMARY KEY,
    BookName VARCHAR(50) NOT NULL,
    Author VARCHAR(50),
    DatePurchased DATE,
    Publisher VARCHAR(50),
    Price DECIMAL(10, 2)
);
