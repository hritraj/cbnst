CREATE TABLE supplier (
    sup_no VARCHAR(5) PRIMARY KEY,
    sup_name VARCHAR(50),
    item_supplied VARCHAR(50),
    item_price DECIMAL(10, 2),
    city VARCHAR(50)
);
